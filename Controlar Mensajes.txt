Contrólalo desde Game.gd con temporizadores. Por ejemplo:

@onready var dialogue_layer = $DialogueLayer

func _ready():
	start_dialogues()
func start_dialogues():
	await get_tree().create_timer(2.0).timeout
	dialogue_layer.show_radio_message("Navegante: Capitana, el sol acaba de colapsar.")

	await get_tree().create_timer(18.0).timeout
	dialogue_layer.show_radio_message("Navegante: Detecto Elemento N en las naves enemigas.")

	await get_tree().create_timer(45.0).timeout
	dialogue_layer.show_radio_message("Navegante: Algo no va bien. El hiperespacio está mezclándose con la realidad.")
